2) ISPISUJEMO SLUZBE OD TRENUTNOG DANA PA DOKLE IMAMO
razlika = todayDate - lastRecordedServiceDate 

ako je danas PON 
- razlika >=8 = > ucitaj nove sluzbe
- razlika 1 = > ucitaj nove sluzbe 
- razlika -6 = > nista

ako je danas UTO
- razlika >= 2 => ucitaj nove sluzbe
- razlika -5 => nista

ako je danas SRI
- razlika >= 3 => ucitaj nove sluzbe
- razlika -4 = > nista

ako je danas CET
- razlika >= 4 = > ucitaj nove 
- razlika -3 => nista

ako je danas PET

 - razlika >= 5
	- ako nisu izasle sluzbe => ucitaj nove
	- ako su izasle sluzbe => ucitaj nove (neces imat petak, subotu i nedjelju ali sam si si kriv)
- razlika -2 (imamo petak, subotu nedjelju) (flagFriday = True)
	- ako nisu izasle sluzbe => nista (loadServices = True i provjeravamo jesu li izasle nove sluzbe u readServices() funkciji)
	- ako su izasle sluzbe => ucitaj nove (imat ces petak, subotu i nedjelju) 
- razlika -9
	=> nista (sluzbe su izasle i imamo ih)

ako je danas SUB
- razlika >= 6 => ucitaj nove (neces imati ovu subotu i nedjelju)
- razlika -1 => ucitaj nove (imat ces subotu i nedjelju)
- razlika -8 => nista

ako je danas NED
- razlika >= 7 => ucitaj nove (neces imat nedjelju)
- razlika 0 => ucitaj nove (imat ces nedjelju)
- razlika -7 => nista
